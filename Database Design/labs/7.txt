CREATE DATABASE IF NOT EXISTS payroll_system;
USE payroll_system;

CREATE TABLE IF NOT EXISTS employees (
    employee_id INT AUTO_INCREMENT PRIMARY KEY,
    employee_name VARCHAR(255) NOT NULL,
    employee_type ENUM('full-time', 'part-time', 'contract') NOT NULL,
    hourly_rate DECIMAL(10, 2),
    monthly_salary DECIMAL(10, 2),
    contract_salary DECIMAL(10, 2)
);

CREATE TABLE IF NOT EXISTS payroll (
    payroll_id INT AUTO_INCREMENT PRIMARY KEY,
    employee_id INT NOT NULL,
    work_hours INT,
    deductions DECIMAL(10, 2),
    bonuses DECIMAL(10, 2),
    month INT NOT NULL,
    year INT NOT NULL,
    FOREIGN KEY (employee_id) REFERENCES employees (employee_id)
);

CREATE TABLE IF NOT EXISTS payroll_reports (
    report_id INT AUTO_INCREMENT PRIMARY KEY,
    employee_id INT NOT NULL,
    total_salary DECIMAL(10, 2),
    month INT NOT NULL,
    year INT NOT NULL,
    FOREIGN KEY (employee_id) REFERENCES employees (employee_id)
);

DELIMITER //

CREATE PROCEDURE calculate_employee_salary()
BEGIN
    DECLARE emp_done BOOLEAN DEFAULT FALSE;
    DECLARE cur_emp_id, cur_work_hours INT;
    DECLARE cur_deductions, cur_bonuses DECIMAL(10,2);
    DECLARE emp_type VARCHAR(20);
    DECLARE hr_rate, mon_salary, cont_salary, cal_salary DECIMAL(10,2);
    DECLARE payroll_cursor CURSOR FOR 
    
    SELECT employee_id, work_hours, IFNULL(deductions, 0), IFNULL(bonuses, 0) 
    FROM payroll WHERE month = 10 AND year = 2023;
    DECLARE CONTINUE HANDLER FOR NOT FOUND SET emp_done = TRUE;

    OPEN payroll_cursor;

    payroll_loop: LOOP
        FETCH payroll_cursor INTO cur_emp_id, cur_work_hours, cur_deductions, cur_bonuses;
        
        IF emp_done THEN
            LEAVE payroll_loop;
        END IF;

		SELECT employee_type, hourly_rate, monthly_salary, contract_salary 
        INTO emp_type, hr_rate, mon_salary, cont_salary
        FROM employees WHERE employee_id = cur_emp_id;

         CASE 
            WHEN emp_type = 'full-time' THEN 
                SET cal_salary = mon_salary + cur_bonuses - cur_deductions;
            WHEN emp_type = 'part-time' THEN 
                SET cal_salary = (hr_rate * cur_work_hours) + cur_bonuses - cur_deductions;
            WHEN emp_type = 'contract' THEN 
                SET cal_salary = cont_salary + cur_bonuses - cur_deductions;
        END CASE;

         INSERT INTO payroll_reports (employee_id, total_salary, month, year)
        VALUES (cur_emp_id, cal_salary, 10, 2023);
    END LOOP;

    CLOSE payroll_cursor;
END //
DELIMITER ;

 CALL calculate_employee_salary();
