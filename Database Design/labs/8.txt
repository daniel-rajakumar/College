DROP DATABASE IF EXISTS library_management;
CREATE DATABASE IF NOT EXISTS library_management;
USE library_management;

-- Create the 'books' table
CREATE TABLE books (
    book_id INT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(255) NOT NULL,
    author VARCHAR(255) NOT NULL,
    isbn VARCHAR(14) NOT NULL,
    copies_available INT NOT NULL
);

-- Create the 'borrowers' table
CREATE TABLE borrowers (
    borrower_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    email VARCHAR(255) NOT NULL
);

-- Create the 'borrowings' table
CREATE TABLE borrowings (
    borrowing_id INT AUTO_INCREMENT PRIMARY KEY,
    book_id INT NOT NULL,
    borrower_id INT NOT NULL,
    checkout_date DATE NOT NULL,
    return_date DATE
);

CREATE PROCEDURE CheckoutBook(IN p_book_id INT, IN p_borrower_id INT)
BEGIN
    DECLARE available_copies INT;

    SELECT copies_available INTO available_copies
    FROM books
    WHERE book_id = p_book_id;
    
    IF available_copies > 0 THEN
        UPDATE books
        SET copies_available = copies_available - 1
        WHERE book_id = p_book_id;

		SELECT p_borrower_id;  

        INSERT INTO borrowings (book_id, borrower_id, checkout_date)
        VALUES (p_book_id, p_borrower_id, CURDATE());
    END IF;
END//


CREATE PROCEDURE ReturnBook(IN p_borrowing_id INT)
BEGIN
    DECLARE book_id INT;

    SELECT book_id INTO book_id
    FROM borrowings
    WHERE borrowing_id = p_borrowing_id
    LIMIT 1;

    UPDATE borrowings
    SET return_date = CURDATE()
    WHERE borrowing_id = p_borrowing_id;

    UPDATE books
    SET copies_available = copies_available + 1
    WHERE book_id = book_id;
END//



CREATE TABLE overdue_books (
    overdue_id INT AUTO_INCREMENT PRIMARY KEY,
    borrowing_id INT,
    overdue_date DATE
);


CREATE TRIGGER CheckOverdueBooks_AfterInsert
AFTER INSERT ON borrowings
FOR EACH ROW
BEGIN
    IF NEW.return_date IS NULL AND CURDATE() > DATE_ADD(NEW.checkout_date, INTERVAL 1 DAY) THEN
        INSERT INTO overdue_books (borrowing_id, overdue_date) VALUES (NEW.borrowing_id, CURDATE());
    END IF;
END//

CREATE TRIGGER CheckOverdueBooks_AfterUpdate
AFTER UPDATE ON borrowings
FOR EACH ROW
BEGIN
    IF NEW.return_date IS NULL AND CURDATE() > DATE_ADD(NEW.checkout_date, INTERVAL 14 DAY) THEN
        INSERT INTO overdue_books (borrowing_id, overdue_date) VALUES (NEW.borrowing_id, CURDATE());
    END IF;
END//

DELIMITER ;

CALL CheckoutBook(6, 6);
